2. The Account Management Header (Header.jsx)
This component handles your Name, Email, Password, and Preferences.

JavaScript
import React, { useState, useEffect } from 'react';
import { User, Moon, Sun, Globe, Lock, Camera, ChevronDown } from 'lucide-react';

const Header = () => {
  const [isOpen, setIsOpen] = useState(false);
  const [theme, setTheme] = useState(localStorage.getItem('theme') || 'light');
  const [lang, setLang] = useState(localStorage.getItem('lang') || 'en');

  // Handle Theme Change
  useEffect(() => {
    document.documentElement.classList.toggle('dark', theme === 'dark');
    localStorage.setItem('theme', theme);
  }, [theme]);

  // Handle Language/RTL
  useEffect(() => {
    document.documentElement.dir = lang === 'ar' ? 'rtl' : 'ltr';
    localStorage.setItem('lang', lang);
  }, [lang]);

  return (
    <header className="flex items-center justify-between px-8 py-4 bg-white dark:bg-slate-800 border-b dark:border-slate-700">
      <div className="text-2xl font-bold text-blue-600">MedFlow</div>

      <div className="relative">
        <button onClick={() => setIsOpen(!isOpen)} className="flex items-center gap-2 group">
          <div className="w-10 h-10 rounded-full bg-blue-100 flex items-center justify-center overflow-hidden border-2 border-transparent group-hover:border-blue-500 transition">
            <User className="text-blue-600" />
          </div>
          <span className="dark:text-white font-medium">Account</span>
          <ChevronDown size={16} className="dark:text-white" />
        </button>

        {isOpen && (
          <div className="absolute right-0 mt-3 w-80 bg-white dark:bg-slate-800 border dark:border-slate-700 shadow-2xl rounded-2xl p-5 z-50">
            {/* Profile Section */}
            <div className="flex flex-col items-center mb-6">
              <div className="relative w-20 h-20 mb-3 group cursor-pointer">
                <div className="w-full h-full rounded-full bg-gray-200 overflow-hidden">
                   <img src="/api/placeholder/150/150" alt="profile" />
                </div>
                <div className="absolute inset-0 bg-black/40 rounded-full opacity-0 group-hover:opacity-100 flex items-center justify-center transition">
                  <Camera className="text-white" size={20} />
                </div>
              </div>
              <input type="text" className="text-center w-full font-bold dark:text-white bg-transparent border-b border-transparent hover:border-gray-300 focus:border-blue-500 outline-none" defaultValue="Dr. User Name" />
              <input type="email" className="text-center w-full text-sm text-gray-500 bg-transparent border-none outline-none" defaultValue="doctor@medflow.com" />
            </div>

            {/* Settings */}
            <div className="space-y-4 pt-4 border-t dark:border-slate-700">
              {/* Theme Toggle */}
              <div className="flex items-center justify-between">
                <div className="flex items-center gap-3 dark:text-gray-200">
                  {theme === 'dark' ? <Moon size={18} /> : <Sun size={18} />}
                  <span>{lang === 'en' ? 'Dark Mode' : 'الوضع الليلي'}</span>
                </div>
                <button 
                  onClick={() => setTheme(theme === 'light' ? 'dark' : 'light')}
                  className="w-10 h-5 bg-gray-300 dark:bg-blue-600 rounded-full relative transition"
                >
                  <div className={`absolute top-1 w-3 h-3 bg-white rounded-full transition-all ${lang === 'ar' ? (theme === 'dark' ? 'right-1' : 'right-6') : (theme === 'dark' ? 'left-6' : 'left-1')}`} />
                </button>
              </div>

              {/* Language Selector */}
              <div className="flex items-center justify-between">
                <div className="flex items-center gap-3 dark:text-gray-200">
                  <Globe size={18} />
                  <span>{lang === 'en' ? 'Language' : 'اللغة'}</span>
                </div>
                <select 
                  value={lang} 
                  onChange={(e) => setLang(e.target.value)}
                  className="text-sm bg-gray-100 dark:bg-slate-700 dark:text-white rounded px-2 py-1 outline-none"
                >
                  <option value="en">English</option>
                  <option value="ar">العربية</option>
                </select>
              </div>

              {/* Password Change */}
              <button className="flex items-center gap-3 w-full text-left py-2 px-1 hover:bg-gray-100 dark:hover:bg-slate-700 rounded transition dark:text-gray-200">
                <Lock size={18} />
                <span>{lang === 'en' ? 'Update Password' : 'تحديث كلمة المرور'}</span>
              </button>
            </div>
          </div>
        )}
      </div>
    </header>
  );
};
